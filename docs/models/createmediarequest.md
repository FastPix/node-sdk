# CreateMediaRequest

## Example Usage

```typescript
import { CreateMediaRequest } from "@fastpix/fastpix-node/models";

let value: CreateMediaRequest = {
  inputs: [
    {
      type: "video",
      url:
        "https://static.fastpix.io/gtv-videos-bucket/sample/ForBiggerJoyrides.mp4",
    },
  ],
  metadata: {
    "key1": "value1",
  },
};
```
